package com.ma.lianbiao;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

public class CombinationSum {
    public static List<List<Integer>> combinationSum(int[] candidates, int target) {
        List<List<Integer>> result = new ArrayList<>();
        Arrays.sort(candidates);
        backtrack(result, new ArrayList<>(), candidates, target, 0);
        return result;
    }

    /**
     * 通过回溯法生成组合。
     *
     * @param result      用于存储生成的组合的列表。
     * @param current     当前正在构建的组合。
     * @param candidates  候选数字的数组。
     * @param target      目标和。
     * @param start       开始考虑候选数字的索引。
     */
    private static void backtrack(List<List<Integer>> result, List<Integer> current, int[] candidates, int target, int start) {
        // 基本情况：如果达到目标和，将当前组合添加到结果中。
        if (target == 0) {
            result.add(new ArrayList<>(current));
            return;
        }

        // 探索所有可能性，从给定索引开始尝试每个候选数字。
        for (int i = start; i < candidates.length && candidates[i] <= target; i++) {
            // 将当前候选数字包含在组合中。
            current.add(candidates[i]);

            // 递归地探索剩余目标和的组合。
            backtrack(result, current, candidates, target - candidates[i], i);

            // 回溯：移除最后添加的元素以探索其他可能性。
            current.remove(current.size() - 1);
        }
    }


    public static void main(String[] args) {
        int[] candidates = {2, 3, 6, 7};
        int target = 7;
        List<List<Integer>> result = combinationSum(candidates, target);
        System.out.println(result);
    }
}
